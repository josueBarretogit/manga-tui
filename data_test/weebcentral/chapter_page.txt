        document.addEventListener('alpine:init', () => {
            Alpine.data('singlePageNavigation', function () {
                const readingStylesWithPage = [
                    'single_page',
                    'double_page',
                    'double_page_v2'
                ];
                return {
                    dark_background: this.$persist(false),
                    gap: this.$persist(true),
                    back_to_top: this.$persist(true),
                    reading_style: this.$persist('long_strip'),
                    reading_direction: this.$persist('left_right'),
                    page: parseInt('1'),
                    bookmarks: new Set(JSON.parse("[]")),
                    image_fit: this.$persist('width'),
                    max_page: parseInt('28'),
                    double_max_page: 0,
                    is_last_chapter: 'False'.toLowerCase() === 'true',
                    is_first_chapter: 'False'.toLowerCase() === 'true',
                    is_prev: 'False'.toLowerCase() === 'true',
                    init() {
                        if (this.page !== 1 && !this.is_prev) {
                            this.reading_style = 'single_page';
                        }
                    },
                    handleClick(element) {
                        const clickX = element.clientX;
                        const sectionWidth = element.currentTarget.getBoundingClientRect().width;

                        const isLeftHalf = clickX < sectionWidth / 2;
                        const isLeftToRight = this.reading_direction === 'left_right';

                        if (isLeftHalf === isLeftToRight) {
                            this.prevPage()
                        } else {
                            this.nextPage()
                        }
                    },
                    handleShortcutKey(event) {
                        if (document.activeElement.tagName == 'TEXTAREA') {
                            return
                        }
                        if (event.key.toLowerCase() === 'g') {
                            this.gap = !this.gap
                        }
                    },
                    prevPage() {
                        if (document.activeElement.tagName == 'TEXTAREA') {
                            return
                        }

                        if (readingStylesWithPage.includes(this.reading_style) && this.page > 1) {
                            this.page--
                            scrollIntoView(this.calculateScrollOffset())
                            return
                        }

                        if (this.is_first_chapter) {
                            return
                        }

                        window.location.href = "https://weebcentral.com/chapters/01JFWESMBYEGFG2BXNX1C2FMVK?is_prev=True"
                    },
                    nextPage() {
                        if (document.activeElement.tagName == 'TEXTAREA') {
                            return
                        }

                        if (readingStylesWithPage.includes(this.reading_style) && this.page < this.max_page) {
                            this.page++
                            scrollIntoView(this.calculateScrollOffset())
                            return
                        }

                        if (this.is_last_chapter) {
                            document.getElementById("last_chapter_modal").showModal()
                            return
                        }

                        window.location.href = "https://weebcentral.com/chapters/01JQ5N6QKKBWVX427RT44K14WV"
                    },
                    is_bookmarked() {
                        return (this.reading_style === "single_page" && this.bookmarks.has(this.page))
                            || (this.reading_style === "long_strip" && this.bookmarks.has(0))
                    },
                    get currentPage() {
                        return this.reading_style === "single_page" ? this.page : 0
                    },
                    load_page() {
                        this.max_page = parseInt(document.getElementById('max_page').value);
                        this.page = parseInt(document.getElementById('current_page').value);
                    },
                    calculateScrollOffset() {
                        if (!this.sticky_top_nav) {
                            return 0
                        }
                        return -document.getElementById("main-nav").offsetHeight
                    }
                }
            });
