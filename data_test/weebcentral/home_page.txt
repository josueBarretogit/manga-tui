                    historyArr: [],

                        this.historyArr = (currentHistoryArr || []).map(history => ({
                            ...history,
                            series_name: this.decodeHTMLEntities(history.series_name)

                    async clearHistory() {
                        await idbKeyval.del("history");
                        this.historyArr = []
